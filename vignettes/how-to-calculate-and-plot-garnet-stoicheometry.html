# Calculating all in a single dataframe

BCH_01<-garnet_atom_units(Gnt_1) %>% 
        garnet_site_occupancy(.) %>% 
        cbind(.,garnet_endmember_fe3(.)) %>% 
        cbind(.,garnet_endmember_fe(.)) %>% 
        cbind(.,garnet_endmember_fe2(.))
